Here is the javascript code that sets up this example: 
BridgeKit.registerNativeEvent('UIApplicationDidBecomeActiveNotification', function () {
    document.getElementById('textUpdate').innerHTML = 'Did become active!';
});

BridgeKit.registerNativeEvent('UIApplicationWillResignActiveNotification', function () {
    document.getElementById('textUpdate').innerHTML = 'Will resign active!';
});
